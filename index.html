        <div class="scoreboard">

            <div class="team">

                <h2>Home</h2>

                <p id="homeScore">0</p>

                <button onclick="incrementScore('home')">+1</button>

            </div>

            <div class="team">

                <h2>Away</h2>

                <p id="awayScore">0</p>

                <button onclick="incrementScore('away')">+1</button>

            </div>

        </div>

        <div class="pattern">

            <p>Current: <span id="currentPattern">A</span></p>

        </div>

    </div>


    <script>

        let homeScore = 0;

        let awayScore = 0;

        let totalScore = 0;

        const pattern = ['A', 'B', 'B', 'A'];


        function incrementScore(team) {

            if (team === 'home') {

                homeScore++;

                document.getElementById('homeScore').textContent = homeScore;

            } else {

                awayScore++;

                document.getElementById('awayScore').textContent = awayScore;

            }

            totalScore++;

            updatePattern();

        }


        function updatePattern() {

            const patternIndex = totalScore % 4;

            document.getElementById('currentPattern').textContent = pattern[patternIndex];

        }

    </script>
